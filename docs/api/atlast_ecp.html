439def wrap(client):
440    """
441    Wrap any supported LLM client with ECP passive recording.
442
443    Supports: Anthropic, OpenAI, AzureOpenAI, Google Gemini, LiteLLM.
444    Auto-detects client type. Handles both regular and streaming calls.
445    Fail-Open: if wrapping fails, returns unchanged client.
446
447    Usage:
448        client = wrap(Anthropic())       # Anthropic
449        client = wrap(OpenAI())          # OpenAI
450        client = wrap(model)             # Google Gemini GenerativeModel
451        litellm = wrap(litellm)          # LiteLLM module
452
453    Streaming works transparently:
454        stream = client.chat.completions.create(messages=[...], stream=True)
455        for chunk in stream:
456            print(chunk)  # No change — chunks arrive at same speed
457        # After stream ends, ECP record is created in background
458
459    Returns: wrapped client (same type, same interface, recording added)
460    """
461    try:
462        class_name = type(client).__name__
463        module_name = type(client).__module__
464
465        # Anthropic
466        if "anthropic" in module_name.lower() or class_name == "Anthropic":
467            return _wrap_anthropic(client)
468
469        # OpenAI / Azure
470        if "openai" in module_name.lower() or class_name in ("OpenAI", "AzureOpenAI"):
471            return _wrap_openai(client)
472
473        # Google Gemini
474        if "google" in module_name.lower() or class_name == "GenerativeModel":
475            if hasattr(client, "generate_content"):
476                return _wrap_gemini(client)
477
478        # LiteLLM (module, not instance)
479        if hasattr(client, "completion") and hasattr(client, "acompletion"):
480            if getattr(client, "__name__", "") == "litellm":
481                return _wrap_litellm(client)
482
483        # Unknown client — return as-is (fail-open)
484        return client
485
486    except Exception:
487        # Wrapping failed — return original client, Agent unaffected
488        return client

 76    ECP's single unified interface.
 77
 78    All adapters call this function. It:
 79      1. Hashes input/output (content never leaves device)
 80      2. Detects behavioral flags passively
 81      3. Chains to previous record
 82      4. Signs with ed25519
 83      5. Saves to local .ecp/

 88    """
 89    try:
 90        identity = _state.identity
 91
 92        # Extract text for flag detection
 93        out_text = _extract_text(output_content)
 94
 95        # Auto-detect retry if not explicitly set
 96        if is_retry is None:
 97            in_hash = hash_content(input_content)
 98            is_retry = _state.check_retry(in_hash)
 99
100        # Passive behavioral flag detection
101        flags = detect_flags(out_text, is_retry=is_retry, latency_ms=latency_ms)
102
103        # Create record (chained to previous)
104        rec = create_record(
105            agent_did=identity["did"],
106            step_type=step_type,

108            out_content=output_content,

117        )
118

120        rec_dict = record_to_dict(rec)
121        save_record(rec_dict, local_summary=local_summary)

110def verify_merkle_proof(record_hash: str, proof: list[dict], merkle_root: str) -> bool:
111    """
112    Verify a Merkle proof for a record against a known Merkle root.
113
114    Args:
115        record_hash: The record's hash (sha256: prefixed)
116        proof: Proof path from build_merkle_proof()
117        merkle_root: The batch's Merkle root (sha256: prefixed)

def build_merkle_proof(all_hashes: list[str], record_hash: str) -> list[dict]: 
 74def build_merkle_proof(all_hashes: list[str], record_hash: str) -> list[dict]:
 75    """
 76    Build a Merkle proof path for a specific record hash.
 77
 78    Args:
 79        all_hashes: All record hashes (sha256: prefixed) in the batch, in order
 80        record_hash: The specific record's hash to prove inclusion of
 81
 82    Returns:
 83        List of {hash, position} steps for verification.
 84        Empty list if record_hash not found in all_hashes.
 85    """
 86    if record_hash not in all_hashes:
 87        return []
 88
 89    hashes = list(all_hashes)
 90    idx = hashes.index(record_hash)
 91    proof = []
 92
 93    while len(hashes) > 1:
 94        if len(hashes) % 2 == 1:
 95            hashes = hashes + [hashes[-1]]
 96
 97        sibling_idx = idx ^ 1
 98        position = "right" if idx % 2 == 0 else "left"
 99        proof.append({"hash": hashes[sibling_idx], "position": position})
100
101        hashes = [
102            _sha256(hashes[i] + hashes[i + 1])
103            for i in range(0, len(hashes), 2)
104        ]
105        idx //= 2
106
107    return proof
